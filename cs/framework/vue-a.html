ж•°жҚ®еҲқе§ӢеҢ–пјҡи§ЈжһҗappдёӢзҡ„жүҖжңүеӯҗиҠӮзӮ№пјҢе°ҶжүҖжңүзҡ„з»‘е®ҡеҖјжӣҝжҚўжҲ–иөӢеҖјдёәmodelйҮҢзҡ„еҖјгҖӮ

ж•°жҚ®еҠ«жҢҒпјҡдёәжңүv-modelеұһжҖ§зҡ„е…ғзҙ ж·»еҠ дәӢд»¶зӣ‘еҗ¬еҷЁпјҢе®һзҺ°view=>modelгҖӮ

и®ўйҳ…/еҸ‘еёғжЁЎејҸпјҡеӣ дёәеӯҳеңЁеӨҡдёӘиҠӮзӮ№зҡ„еҶ…е®№еҜ№еә”еҗҢдёҖдёӘеұһжҖ§еҖјпјҢеҸҜд»ҘдёәжҜҸдёӘиҠӮзӮ№и®ҫзҪ®дёҖдёӘи®ўйҳ…иҖ…еҜ№иұЎпјҢе°ҶиҝҷдёӘи®ўйҳ…иҖ…еҜ№иұЎеҠ е…ҘиҝҷдёӘеұһжҖ§зҡ„и®ўйҳ…иҖ…еҲ—иЎЁдёӯгҖӮдёҺжӯӨеҗҢж—¶пјҢеұһжҖ§зҡ„и®ҝй—®еҷЁжҲҗдёәеҸ‘еёғиҖ…пјҢд»ҺиҖҢе®һзҺ°model=>viewsгҖӮ

з”ұдәҺж•°жҚ®еҠ«жҢҒйҮҮз”Ёзҡ„жҳҜObject.defineProperty()пјҢдёҖдәӣеҜ№еұһжҖ§зҡ„ж“ҚдҪңж— жі•жӢҰжҲӘпјҢеҰӮйҖҡиҝҮдёӢж Үж–№ејҸдҝ®ж”№ж•°з»„ж•°жҚ®жҲ–иҖ…з»ҷеҜ№иұЎж–°еўһеұһжҖ§гҖӮ

Vue3.0дҝ®еӨҚдәҶиҝҷй—®йўҳпјҢе…¶дҪҝз”ЁProxy еҜ№еҜ№иұЎиҝӣиЎҢд»ЈзҗҶпјҡзӣ‘еҗ¬зҡ„иҢғеӣҙжӣҙе№ҝпјҢе°ҒиЈ…жӣҙеҘҪгҖӮ

еҸҳеҢ–дҫҰеҗ¬ вҖӢ

дҫҰжөӢж•°жҚ®зҡ„еҸҳеҢ–гҖӮеҪ“ж•°жҚ®еҸҳеҢ–ж—¶пјҢжӣҙж–°и§ҶеӣҫпјҢеҚіmodel=>viewиҝҮзЁӢгҖӮvueдё»иҰҒйҖҡиҝҮ[[front-end#ж•°жҚ®з»‘е®ҡ|ж•°жҚ®еҠ«жҢҒ]]ж–№ејҸе®һзҺ°гҖӮдё»иҰҒжңүеҮ зӮ№пјҡ

  • иЎҢдёәжӢҰжҲӘпјҡйҖҡиҝҮObject.definePropertyеҜ№еҜ№иұЎзҡ„еұһжҖ§зҡ„getterе’ҢsetterиЎҢдёәиҝӣиЎҢжӢҰжҲӘгҖӮ

  • дҫқиө–ж”¶йӣҶпјҡж”¶йӣҶйӮЈдәӣдҪҝз”ЁиҜҘж•°жҚ®зҡ„ең°ж–№пјҢеңЁgetterдёӯиҝӣиЎҢпјҢе°Ҷе…¶ж”ҫе…ҘдёҖдёӘdepеҜ№иұЎпјҢе…¶дёӯжңүеҢ…еҗ«дҫқиө–зҡ„ж•°з»„subsгҖӮ

  • дҫқиө–жҠҪиұЎпјҡеҜ№йӮЈдәӣдҪҝз”Ёж•°жҚ®зҡ„ең°ж–№зҡ„жҠҪиұЎпјҢз§°дёәwatcherпјҢеҢ…еҗ«ж•°жҚ®еҸҳеҢ–ж—¶зҡ„ж“ҚдҪңпјҢеҰӮеӣһи°ғжҲ–и§Ҷеӣҫжӣҙж–°гҖӮж–°е»әж—¶дјҡиҮӘеҠЁиў«ж·»еҠ е…ҘsubsгҖӮ

  • дҫҰжөӢе°ҒиЈ…пјҡе®һзҺ°дёҖдёӘObserverзұ»пјҢеҜ№дёҖдёӘеҜ№иұЎзҡ„жүҖжңүеұһжҖ§йҖ’еҪ’ең°иҝӣиЎҢдҫҰжөӢгҖӮ

Object вҖӢ

еҹәжң¬е®һзҺ°

js
class Observer {
  constructor(obj) {
    this.obj = obj
    if (!Array.isArray(obj)) this.walk(obj);
  }
  walk(obj) {
    for (let key of Object.keys()) {
      defineReactive(obj, key, obj[key])
    }
  }
}

function defineReactive(data, key, val) {
  if (typeof val === 'object') new Observer(val);
  
  let dep = new Dep()
  Object.defineProperty(data, key, {
    enumerable: true,
    configurable: true,
    get() {
      dep.depend() // ж”¶йӣҶдҫқиө–
      return val
    },
    set(newVal) {
      if (val === newVal) return;
      val = newVal
      dep.notify() // и§Ҷеӣҫжӣҙж–°
    }
  })
}

йҮҮз”ЁObject.definePropertyпјҢеҜ№еҜ№иұЎең°ж·»еҠ жҲ–еҲ йҷӨеұһжҖ§иЎҢдёәж— жі•иҝӣиЎҢжӢҰжҲӘгҖӮ

ж•°з»„ вҖӢ

з”ұдәҺеҜ№иұЎеўһеҲ еұһжҖ§зҡ„иЎҢдёәж— жі•жӢҰжҲӘпјҢеӣ жӯӨеҜ№дәҺж•°з»„иҝҷзұ»йў‘з№ҒиҝӣиЎҢеўһеҲ е…ғзҙ зҡ„еҜ№иұЎиҖҢиЁҖпјҢе№¶дёҚйҖӮеҗҲзӣҙжҺҘеҜ№е…¶еұһжҖ§пјҢеҚізҙўеј•иҝӣиЎҢдҫҰжөӢгҖӮеӣ иҖҢйҮҮеҸ–еҜ№ж•°з»„ж–№жі•зҡ„жӢҰжҲӘж–№ејҸпјҢеҚіеҜ№еҺҹз”ҹж–№жі•зҡ„иҰҶзӣ–гҖӮ

дёӨиҖ…дёҚиғҪе…ұеӯҳпјҢеӣ дёәеҺҹз”ҹж•°з»„ж–№жі•зҡ„еҶ…йғЁеҗ«жңүеҜ№еұһжҖ§зҡ„и®ҝй—®дёҺиөӢеҖјгҖӮеҗҢж—¶еӯҳеңЁж—¶и®ҝй—®еҷЁе°Ҷиў«и§ҰеҸ‘дёӨж¬ЎгҖӮ

з”ұдәҺдҫҰжөӢжҳҜеңЁж•ҙдёӘж•°з»„дёҠзҡ„пјҢиҖҢйқһеҚ•дёӘеұһжҖ§пјҢеӣ жӯӨDepеҜ№иұЎж”ҫеңЁObserverдёӯгҖӮ

еҜ№дәҺж•°з»„еҶ…зҡ„еҜ№иұЎжҲ–ж•°з»„пјҲдёҚжҳҜдҫҰеҗ¬зҙўеј•пјүпјҢеҗҢж ·йңҖиҰҒиҝӣиЎҢдҫҰеҗ¬гҖӮ

жҜҸдёӘwatcherеҜ№еә”дёҖдёӘз»„д»¶пјҢеҸҜи§ӮеҜҹеӨҡдёӘж•°жҚ®пјҢеӣ жӯӨDepе’ҢwatcherжҳҜеӨҡеҜ№еӨҡе…ізі»гҖӮ

ж–°е»әwatcherж—¶пјҢе…¶иў«иҮӘеҠЁж·»еҠ е…ҘеҜ№еә”зҡ„subsеҲ—иЎЁдёӯгҖӮ

class Watcher {
  constructor(vm, expOrFn, cb) {
    this.vm = vm
    this.deps = [] // еҸҜи§ӮеҜҹеӨҡдёӘж•°жҚ®
    this.depIds = new Set()
    // жү§иЎҢthis.getter()пјҢдјҡиҜ»еҸ–жүҖжңүеұһжҖ§зҡ„еҖјпјҢеҮҪж•°еҶ…зҡ„жҲ–'a.b.c'
    this.getter = isFunction(expOrFn) ? expOrFn : parsePath(expOrFn)
    this.cb = cb
    this.value = this.get()
  }
  get() { // иҜ»еҸ–еұһжҖ§еҖј
    window.target = this
    let value = this.getter.call(this.vm, this.vm)
    window.target = undefined
    return value
  }
  update() { // жү§иЎҢеӣһи°ғ
    const oldValue = this.value
    this.value = this.get()
    this.cb.call(this.vm, this.value, oldValue)
  }
}

йҖҡиҝҮJSеҜ№иұЎзҡ„зј“еӯҳж“ҚдҪңпјҢеҸҜд»ҘеҮҸе°‘еҜ№DOMзҡ„ж“ҚдҪңж¬Ўж•°гҖӮеӨ§иҮҙиҝҮзЁӢпјҡ

  1. е°ҶDOMз”ЁJSеҜ№иұЎиЎЁзӨәгҖӮдё»иҰҒжҳҜеӯҳеӮЁиҠӮзӮ№зҡ„зұ»еһӢпјҢеұһжҖ§пјҢеӯҗиҠӮзӮ№гҖӮ
  2. DOMзҠ¶жҖҒеҸҳжӣҙж—¶пјҢйҮҚж–°жһ„йҖ дёҖжЈөж–°зҡ„еҜ№иұЎж ‘гҖӮз”Ёdiffз®—жі•жүҫеҮәж–°ж—§дёӨж ‘зҡ„дёҚеҗҢгҖӮ
  3. ж №жҚ®еҫ—еҲ°зҡ„дёҚеҗҢеҜ№DOMиҝӣиЎҢзӣёеә”зҡ„ж“ҚдҪңгҖӮ

vueж–№жЎҲ вҖӢ

еҸҜд»ҘеҜ№жҜҸдёӘдҪҝз”ЁдәҶж•°жҚ®зҡ„иҠӮзӮ№з»‘е®ҡдёҖдёӘwatcherпјҢиҝӣиҖҢе®һзҺ°иҠӮзӮ№дёҺж•°жҚ®зҡ„з»‘е®ҡпјҢдҪҶиҝҷж ·еҒҡеҶ…еӯҳејҖй”ҖеҫҲеӨ§гҖӮ

vueйҖҡиҝҮеј•е…ҘиҷҡжӢҹDOMпјҢйҮҮеҸ–дәҶдёҖз§ҚжҠҳдёӯзҡ„ж–№жЎҲпјҡжҜҸдёӘз»„д»¶еҜ№еә”дёҖдёӘwatcherгҖӮеҪ“ж•°жҚ®еҸ‘з”ҹеҸҳеҢ–ж—¶пјҢжүҖжңүдҫқиө–иҜҘж•°жҚ®зҡ„з»„д»¶дјҡйҖҡиҝҮиҷҡжӢҹDOMеҺ»иҝӣиЎҢжҜ”еҜ№пјҢиҝӣиҖҢжӣҙж–°жёІжҹ“йӮЈдәӣйңҖиҰҒеҸҳеҢ–зҡ„иҠӮзӮ№гҖӮ

diffз®—жі• вҖӢ

иҷҡжӢҹDOMзҡ„еә”з”Ёе…ій”®еңЁдәҺdiffз®—жі•пјҢе…¶жүҫеҮәж–°ж—§дёӨйў—иҷҡжӢҹDOMж ‘е·®ејӮгҖӮvueйҮҮеҸ–зҡ„ж–№ејҸдёәеҗҢеұӮжҜ”иҫғгҖӮ

жЁЎжқҝзј–иҜ‘ вҖӢ

е°ҶжЁЎжқҝзј–иҜ‘пјҢжңҖз»Ҳи§ЈжһҗдёәдёҖдёӘжёІжҹ“еҮҪж•°гҖӮ

дё»иҰҒжңүдёүйғЁеҲҶеҶ…е®№пјҡ

  • е°ҶжЁЎжқҝи§ЈжһҗдёәжҠҪиұЎиҜӯжі•ж ‘ASTгҖӮ

  • йҒҚеҺҶASTж Үи®°йқҷжҖҒиҠӮзӮ№гҖӮ

  • дҪҝз”ЁASTз”ҹжҲҗд»Јз Ғеӯ—з¬ҰдёІпјҢдҪңдёәжёІжҹ“еҮҪж•°зҡ„еҸӮж•°гҖӮ

дёӢйқўжҳҜдёҖдёӘдҫӢеӯҗгҖӮ

жЁЎжқҝ

html
<p title="Berwin" @click="c">hello {{name}}</p>

з”ҹжҲҗзҡ„д»Јз Ғеӯ—з¬ҰдёІ

js
with (this) {
  return _c( // createElement
    'p',
    {
      attrs: { "title": "Berwin" },
      on: { "click": c }
    },
    [_v("hello" + _s(name))] // _v: createTextVNode _s: toString
  )
}

жү§иЎҢжёІжҹ“еҮҪж•°пјҢз”ҹжҲҗVNodeгҖӮ

еҪ“и§Јжһҗзҡ„иҠӮзӮ№дёәз»„д»¶иҠӮзӮ№ж—¶пјҢдјҡе°Ҷе…¶ж ҮзӯҫеұһжҖ§дёҠи§ЈжһҗеҮәзҡ„ж•°жҚ®еҪ“дҪңеҸӮж•°дј йҖ’з»ҷеӯҗз»„д»¶гҖӮ

ејӮжӯҘжӣҙж–° вҖӢ

vueж•°жҚ®зҡ„еҸҳжӣҙдёҚдјҡз«ӢеҚіжү§иЎҢDOMзҡ„жӣҙж”№пјҢиҝҷдәӣеҸҳжӣҙзҠ¶жҖҒпјҲwatcherпјүиў«ж”ҫе…ҘдёҖдёӘжӣҙж–°йҳҹеҲ—дёӯгҖӮиҝҷз§Қзӯ–з•ҘеҸҜд»ҘеҮҸе°‘еҜ№DOMзҡ„ж“ҚдҪңгҖӮеӨ§иҮҙиҝҮзЁӢдёәпјҡ

  1. дёҖејҖе§ӢпјҢж•°жҚ®еҸҳжӣҙж—¶пјҢйҖҡиҝҮnextTickе°ҶеҜ№еә”зҡ„watcherж”ҫе…ҘйҳҹеҲ—queueпјҢqueueиў«ж”ҫе…Ҙжӣҙж–°йҳҹеҲ—callbacksдёӯпјҢеҗҢж—¶ејҖеҗҜдёҖдёӘе®ҸжҲ–еҫ®д»»еҠЎеҺ»жү§иЎҢжӣҙж–°йҳҹеҲ—гҖӮ

зӣҙеҲ°иҜҘд»»еҠЎжү§иЎҢд№ӢеүҚзҡ„иҝҷж®өж—¶й—ҙпјҢеҸ‘з”ҹзҡ„ж•°жҚ®еҸҳжӣҙиў«еҺ»йҮҚең°еҺӢе…ҘйҳҹеҲ—пјҢдҪҶдёҚдјҡеҶҚејҖеҗҜе®ҸжҲ–еҫ®д»»еҠЎгҖӮ

  1. apiи°ғз”ЁnextTickж—¶дјҡе°Ҷе…¶еӣһи°ғеҮҪж•°еҺӢе…ҘcallbacksгҖӮ

  2. жү§иЎҢwatcherе®ҢжҲҗж—¶пјҢжӯӨж—¶еҺҹз”ҹDOMзҠ¶жҖҒеҸ‘з”ҹжӣҙж”№пјҢдҪҶжңӘеңЁйЎөйқўжҳҫзӨәгҖӮ

  3. еҪ“еҪ“еүҚдәӢд»¶еҫӘзҺҜз»“жқҹж—¶пјҲеҚіеҫ®йҳҹеҲ—зҡ„д»»еҠЎе…ЁйғЁиў«жү§иЎҢпјүпјҢжөҸи§ҲеҷЁжёІжҹ“йЎөйқўпјҢжӯӨж—¶жүҖжңүзҠ¶жҖҒеҸҳжӣҙеңЁйЎөйқўдёҠз”ҹж•ҲгҖӮ

nextTick вҖӢ

д»ҺдёҠйқўеҸҜд»Ҙеҫ—зҹҘпјҢnextTickзҡ„еӣһи°ғеҮҪж•°еҸ‘з”ҹеңЁDOMжӣҙж–°д№ӢеҗҺпјҢж—ўжңүпјҡ

js
vm.message = 'new message' // жӣҙж”№ж•°жҚ®
vm.$el.textContent === 'new message' // false
Vue.nextTick(function () {
  vm.$el.textContent === 'new message' // true
})

жҳҫз„¶пјҢж №жҚ®е…¶еҺҹзҗҶпјҢеҸҜд»ҘдҪҝnextTickжӣҝжҚўжҲҗsetTimeoutпјҢд»ҺиҖҢиҫҫеҲ°дёҖж ·зҡ„ж•ҲжһңгҖӮ

е®Ҹ/еҫ®д»»еҠЎ вҖӢ

nextTickй»ҳи®ӨйҮҮз”Ёеҫ®д»»еҠЎпјҢеҰӮжһңдёҚиЎҢпјҢеҲҷйҖҖдёәе®Ҹд»»еҠЎгҖӮе…·дҪ“зҡ„йЎәеәҸдёәпјҡ

promise.then -> MutationObserver -> setImmediate -> setTimeout

з”ҹе‘Ҫе‘Ёжңҹ вҖӢ

еҲқе§ӢеҢ– вҖӢ

vueзӨәдҫӢеҲӣе»әж—¶пјҢдјҡиҝӣиЎҢдёҖзі»еҲ—еҲқе§ӢеҢ–пјҢдё»иҰҒжҳҜпјҡ

js
// еҗҲе№¶options
vm.$options = mergeOptions() 
// еҗ‘е®һдҫӢжҢӮиҪҪеұһжҖ§пјҢеҰӮ$parent, $_wactherзӯү
initLifecycle(vm) 
// е°ҶеңЁзҲ¶з»„д»¶дҪҝз”Ёзҡ„v-onжіЁеҶҢзҡ„дәӢд»¶ж·»еҠ еҲ°еӯҗз»„д»¶зҡ„дәӢд»¶зі»з»ҹ
initEvents(vm) 
initRender(vm) // render зӣёе…ізҡ„еұһжҖ§пјҢ_c, $attrs
callHook(vm, 'beforeCreate') 
// еңЁdata/propsеүҚеҲқе§ӢеҢ–inject, иҺ·еҸ–зҲ¶з»„д»¶зҡ„provideеҖје№¶еҲқе§ӢеҢ–
initInjections(vm) 
initState(vm)
initProvide(vm) // еңЁdata/propsеҗҺеҲқе§ӢеҢ–provide
callHook(vm, 'created')
if (vm.$options.el) {
vm.$mount(vm.$options.el)
}

д»Ҙ $ ејҖеӨҙзҡ„еұһжҖ§жҳҜжҸҗдҫӣз»ҷз”ЁжҲ·дҪҝз”Ёзҡ„пјҢд»Ҙ _ ејҖеӨҙзҡ„еұһжҖ§жҳҜжҸҗдҫӣз»ҷеҶ…йғЁдҪҝз”Ёзҡ„гҖӮ

inject&provide вҖӢ

зҘ–е…Ҳз»„д»¶йҖҡиҝҮprovideжіЁе…Ҙж•°жҚ®пјҢеӯҗеӯҷз»„д»¶еҸҜд»ҘйҖҡиҝҮinjectиҝҷдәӣж•°жҚ®гҖӮ

injectеңЁdataеүҚеҲқе§ӢеҢ–пјҢиҜҙжҳҺеҸҜд»ҘеңЁdataж•°жҚ®дёӯдҪҝз”Ёinjectзҡ„ж•°жҚ®гҖӮ

state вҖӢ

еҲқе§ӢеҢ–дё»иҰҒзҡ„ж•°жҚ®еұһжҖ§пјҡprops -> methods -> data -> computed -> watchгҖӮ

props е°ҶзҲ¶з»„д»¶зҡ„propsж•°жҚ®дј з»ҷеӯҗз»„д»¶и§„ж јеҢ–зҡ„propsеұһжҖ§пјҢе°ҶжңҖз»Ҳзҡ„ж•°жҚ®дҝқеӯҳеңЁvm._propsдёӯпјҢйҖҡиҝҮд»ЈзҗҶе®һзҺ°vm.xи®ҝй—®vm._props.xгҖӮ

methods жЈҖжҹҘжҳҜеҗҰеҗҲжі•пјҢйҒҚеҺҶе…¶еұһжҖ§пјҢиөӢеҖјз»ҷvmгҖӮ

computed дё»иҰҒзү№зӮ№жҳҜжңүж•°жҚ®зј“еӯҳ(еӯҳеңЁwatcherдёӯ)пјҢдҪҝз”ЁwatcherжқҘи§ӮеҜҹе…¶дёӯзҡ„ж•°жҚ®еҸҳеҢ–гҖӮ

иҝҷж ·зҡ„е®һзҺ°жңүдёӘй—®йўҳпјҡеҰӮжһңи®Ўз®—еұһжҖ§дёӯзҡ„зҠ¶жҖҒеҸ‘з”ҹдәҶеҸҳеҢ–пјҢдҪҶи®Ўз®—еұһжҖ§зҡ„еҖје№¶жІЎжңүеҸҳпјҢжӯӨж—¶д»Қз„¶дјҡйҮҚж–°жёІжҹ“з»„д»¶гҖӮ

ж”№иҝӣзҡ„ж–№жЎҲжҳҜпјҡи®Ўз®—еұһжҖ§зҡ„watcherеҲӨж–ӯжңҖз»ҲеҖјжҳҜеҗҰж”№еҸҳпјҢжқҘеҶіе®ҡжҳҜеҗҰйҖҡзҹҘз»„д»¶зҡ„watcherгҖӮ

з”ҹе‘Ҫй’©еӯҗ вҖӢ

з»„д»¶д»ҺеҲӣе»әеҲ°й”ҖжҜҒзҡ„зҡ„иҝҮзЁӢпјҢдё»иҰҒеҢ…жӢ¬пјҡејҖе§ӢеҲӣе»әгҖҒеҲқе§ӢеҢ–ж•°жҚ®гҖҒзј–иҜ‘жЁЎжқҝгҖҒжҢӮиҪҪDomвҶ’жёІжҹ“гҖҒжӣҙж–°вҶ’жёІжҹ“гҖҒй”ҖжҜҒгҖӮ

VueжҸҗдҫӣдәҶз»„д»¶еңЁз”ҹе‘Ҫе‘Ёжңҹеҗ„дёӘйҳ¶ж®өзҡ„й’©еӯҗеҮҪж•°пјҢд»ҘжӯӨеңЁеҗ„дёӘйҳ¶ж®өе®һзҺ°дёҖдәӣж“ҚдҪңгҖӮ

beforeCreateпјҡ е®һдҫӢеҲқе§ӢеҢ–еүҚиў«и§ҰеҸ‘пјҢи®ҝй—®дёҚеҲ°dataж•°жҚ®гҖӮ

createdпјҡеҲӣе»әе®һдҫӢпјҢе®ҢжҲҗж•°жҚ®еҲқе§ӢеҢ–еҗҺгҖӮеҸҜи®ҝй—® dataгҖҒmethodsпјҢдҪҶи®ҝй—®дёҚеҲ° $el еұһжҖ§гҖӮ

дёҖиҲ¬еңЁжӯӨеҒҡдёҖдәӣйЎөйқўеҲқе§ӢеҢ–зҡ„е·ҘдҪңпјҢжҜ”еҰӮajax иҜ·жұӮж•°жҚ®гҖӮ

beforeMountпјҡжЁЎжқҝи§Јжһҗе®ҢжҲҗгҖӮ

mountedпјҡз»„д»¶е…ғзҙ еҲӣе»әе®ҢжҲҗпјҢеҮҶеӨҮжҢӮиҪҪгҖӮ

е№¶дёҚдёҖе®ҡж·»еҠ еҲ°дәҶDOMпјҢеҸҜд»Ҙз”ЁnextTickжқҘдҝқиҜҒиҝҷдёҖзӮ№гҖӮ

beforeUpdateпјҡ ж•°жҚ®жӣҙж–°е°ҶиҰҒеҜ№DOMеҒҡдёҖдәӣжӣҙж”№ж—¶гҖӮ

еҜ№еҸҜиғҪдјҡиў«з§»йҷӨзҡ„е…ғзҙ еҒҡдёҖдәӣж“ҚдҪңпјҢжҜ”еҰӮз§»йҷӨдәӢд»¶зӣ‘еҗ¬еҷЁгҖӮ

updatedпјҡеңЁDOMзҡ„жӣҙж”№еҸҠжёІжҹ“е®ҢжҜ•еҗҺгҖӮ

beforeDestroyпјҡз»„д»¶еҚіе°Ҷиў«й”ҖжҜҒе№¶дё”д»ҺDOMдёҠз§»йҷӨж—¶гҖӮ

й”ҖжҜҒе®ҡж—¶еҷЁгҖҒи§Јз»‘е…ЁеұҖдәӢд»¶зӯүгҖӮ

destroyedпјҡз»„д»¶иў«й”ҖжҜҒеҗҺгҖӮ

е®һдҫӢдёӯзҡ„жүҖжңүдёңиҘҝйғҪдјҡи§ЈйҷӨз»‘е®ҡпјҢеҢ…жӢ¬дәӢд»¶зӣ‘еҗ¬еҷЁпјҢжүҖжңүзҡ„еӯҗе®һдҫӢгҖӮ

дҪҝз”Ё keep-aliveж—¶пјҢиҝҳжңүдёӨдёӘй’©еӯҗеҮҪж•°пјҡеҲҶеҲ«жҳҜ activated е’Ң deactivated гҖӮз”Ё keep-alive еҢ…иЈ№зҡ„з»„д»¶еңЁеҲҮжҚўж—¶дёҚдјҡиҝӣиЎҢй”ҖжҜҒпјҢиҖҢжҳҜзј“еӯҳеҲ°еҶ…еӯҳдёӯе№¶жү§иЎҢ deactivated й’©еӯҗеҮҪж•°пјҢе‘Ҫдёӯзј“еӯҳжёІжҹ“еҗҺдјҡжү§иЎҢ actived й’©еӯҗеҮҪж•°гҖӮ

зҲ¶еӯҗз»„д»¶ вҖӢ

зҲ¶еӯҗй—ҙзҡ„з”ҹе‘Ҫе‘Ёжңҹжү§иЎҢйЎәеәҸпјҡеҜ№дәҺжҹҗдёӘиҝҮзЁӢпјҢе…Ҳжү§иЎҢзҲ¶е…ғзҙ зҡ„еүҚзҪ®й’©еӯҗпјҢеҶҚжү§иЎҢеӯҗе…ғзҙ зҡ„ж•ҙеҘ—й’©еӯҗпјҢжңҖеҗҺжү§иЎҢзҲ¶е…ғзҙ зҡ„й’©еӯҗгҖӮ

  • еҠ иҪҪжёІжҹ“иҝҮзЁӢ
